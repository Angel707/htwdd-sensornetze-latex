\newcommand{\prosymbol}{%
	\raisebox{-.1\baselineskip}{%
		\begin{tikzpicture}%
			\draw [line width=.1\baselineskip] (-.25\baselineskip,0) -- (.25\baselineskip,0);%
			\draw [line width=.1\baselineskip] (0,-.25\baselineskip) -- (0,.25\baselineskip);%
		\end{tikzpicture}%
	}%
	}

		\end{tikzpicture}%
	}%
	}
